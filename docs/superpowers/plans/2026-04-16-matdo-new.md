    hbm_bytes: int = 80 * 1024**3
    dram_bytes: int = 512 * 1024**3

@dataclass
class ErrorCoefficients:
    alpha: float = 0.015
    beta: float = 2.0
    gamma: float = 0.10
    delta: float = 0.005
    epsilon: float = 0.002
    zeta: float = 0.35
    eta: float = 0.5


@dataclass
class MATDOConfig:
    hardware: HardwareBudgets = field(default_factory=HardwareBudgets)
    coeffs: ErrorCoefficients = field(default_factory=ErrorCoefficients)
    r_options: tuple[int, ...] = (2, 4, 8)
    r_min: int = 2
    e_max: int = 128_000
    e0: float = 10_000.0
    s: int = 4
    n_block: int = 1024
    d_model: int = 4096
    c_unit: float = 1024.0

```python
# MATDO-new/matdo_new/core/policy.py
from __future__ import annotations

from dataclasses import dataclass

from .config import MATDOConfig


@dataclass
class RuntimeObservation:
    rho: float
    rho_dram: float
    target_error: float


@dataclass
class PolicyDecision:
    R: int
    M: int
    T: int
    E: int
    is_arbitrage: bool


def solve_policy(cfg: MATDOConfig, observation: RuntimeObservation) -> PolicyDecision:
    arbitrage_rhs = cfg.coeffs.eta / max(cfg.e_max * observation.target_error, 1e-12)
    is_arbitrage = observation.rho >= 0.93 and cfg.coeffs.zeta > arbitrage_rhs
    e_value = cfg.e_max if is_arbitrage else 0
    m_value = max(1, int(cfg.hardware.hbm_bytes * (1.0 - observation.rho) / (cfg.n_block * cfg.r_min * cfg.c_unit)))
    t_value = min(cfg.t_max_hard, max(1, int((cfg.coeffs.gamma / max(observation.target_error, 1e-6)) ** 2)))
    return PolicyDecision(R=cfg.r_min, M=m_value, T=t_value, E=e_value, is_arbitrage=is_arbitrage)
```

```python
# MATDO-new/matdo_new/core/error_model.py
from __future__ import annotations

import math

from .config import MATDOConfig


def compute_engram_compensation(cfg: MATDOConfig, e_value: int) -> float:
    if e_value <= 0:
        return 1.0
    return 1.0 - cfg.coeffs.zeta * (1.0 - math.exp(-e_value / cfg.e0))


def estimate_error(cfg: MATDOConfig, r_value: int, m_value: int, t_value: int, e_value: int) -> float:
    coeffs = cfg.coeffs
    f_e = compute_engram_compensation(cfg, e_value)
    retrieval = coeffs.eta / e_value if e_value > 0 else 0.0
    return (
        coeffs.alpha * (2 ** (-2 * r_value))
        + (coeffs.beta * f_e) / max(m_value * cfg.s, 1)
        + coeffs.gamma / math.sqrt(max(t_value, 1))
        + coeffs.delta * (2 ** (-2 * r_value)) / max(m_value, 1)
        + coeffs.epsilon * math.log(max(m_value, 2)) / max(t_value, 1)
        + retrieval
    )
```

# MATDO-new/matdo_new/runtime/decode.py
from __future__ import annotations

import torch


def decode_one_token(*, backend, state, last_token: torch.Tensor):
    logits, state = backend.forward_step(last_token, state)
    next_token = torch.argmax(logits[:, -1, :], dim=-1, keepdim=True)
    state.input_ids = torch.cat([state.input_ids, next_token], dim=1)
    state.sequence_length += 1
    return next_token, state
```

# MATDO-new/matdo_new/runtime/generation.py
from __future__ import annotations

import torch

from .decode import decode_one_token
from .prefill import prefill_prompt


def generate_tokens(*, backend, input_ids: torch.Tensor, max_new_tokens: int) -> torch.Tensor:
    state = prefill_prompt(backend=backend, input_ids=input_ids)
    last_token = input_ids[:, -1:]
    for _ in range(max_new_tokens):
        next_token, state = decode_one_token(backend=backend, state=state, last_token=last_token)
        last_token = next_token
    return state.input_ids
```
